Вместо типа в цикле for можно использовать auto

#include <iostream>  /* библиотека для ввода и вывода */
#include <algorithm> /* библиотека для подключения алгоритмов */
#include <vector>
using namespace std;
	int main () {
		vector<int> nums = {1,5,3,4,5};
		int q = count(begin(nums), end(nums),5);
		cout << q;

	return 0;
}
Вывод: 2

#include <iostream>  /* библиотека для ввода и вывода */
#include <algorithm> /* библиотека для подключения алгоритмов */
#include <vector>
using namespace std;
	int main () {
		vector<int> nums = {1,5,3,4,5};
		sort(begin(nums), end(nums));
		for( auto c: nums){
			cout << c << " ";
		}

	return 0;
}

ВИДЕО УСЛОВИЯ ЦИКЛЫ КОММЕНТАРИИ: КРАТКИЙ ОБЗОР

#include <iostream>  /* библиотека для ввода и вывода */
#include <algorithm> /* библиотека для подключения алгоритмов */
#include <vector>
using namespace std;
	int main () {
		int n = 5;
		int sum = 0;
		for (int i = 1 ; i <= n ;++i ){
			sum += i;
		}
		cout << sum;
	return 0;
}

Вывод:15

#include <iostream>  /* библиотека для ввода и вывода */
#include <algorithm> /* библиотека для подключения алгоритмов */
#include <vector>
using namespace std;
	int main () {
		int n = 5;
		// комментарий
		int sum = 0;
		int i = 1;
		while (i <= n){
			sum += i;
			i++;
		}

		cout << sum;
	return 0;
}

Вывод: 15
